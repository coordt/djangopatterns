        fields = [cat.split('.')[2] for cat in registry if registry[cat] == model]
        # check each field to see if already defined
        for cat in fields:
            for k,v in fieldsets:
                if cat in v['fields']:
                    fields.remove(cat)
        # if there are any fields left, add them under the categories fieldset
        if len(fields) > 0:

                    'fields': fields
